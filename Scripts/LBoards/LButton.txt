#alias 4: TYPE_LBUTTON


sub ObjectMain
	if Engine.Message==2
		Object.Value1=0
	endif
	TempValue0=Object.XPos
	TempValue0>>=16
	TempValue1=TempValue0
	TempValue2=Object.YPos
	TempValue2>>=16
	TempValue3=TempValue2
	switch Object.PropertyValue
	case 0
		TempValue1+=64
		TempValue2-=8
		TempValue3+=24
		break
	case 1
		TempValue1+=48
		TempValue2-=8
		TempValue3+=24
		break
	case 2
		TempValue1+=48
		TempValue2-=8
		TempValue3+=24
		break
	case 3
		TempValue0-=8
		TempValue1+=32
		TempValue2-=8
		TempValue3+=40
		break
	case 4
		TempValue0-=8
		TempValue1+=32
		TempValue2-=8
		TempValue3+=40
		break

	endswitch
	if Object[0].State==2
		CheckTouchRect(TempValue0,TempValue2,TempValue1,TempValue3)
	else
		CheckResult=-1
	endif
	if CheckResult>-1
		Object.Value1=1
	else
		CheckTouchRect(0,0,Screen.XSize,Screen.YSize)
		if CheckResult<0
			if Object.Value1==1
				switch Object.PropertyValue
				case 0
					Object[0].State=4
					StopMusic()
					PlaySfx(27,0)
					break
				case 1
					Object[0].State=3
					Object[32].State=10
					PlaySfx(23,0)
					break
				case 2
					Object[0].State=3
					Object[32].State=7
					PlaySfx(23,0)
					break
				case 3
					Object[0].State=3
					Object[32].State=4
					PlaySfx(23,0)
					break
				case 4
					Object[0].State=3
					Object[32].State=1
					PlaySfx(23,0)
					break

				endswitch
				Object.Value1=0
			endif
		else
			Object.Value1=0
		endif
	endif
endsub


sub ObjectDraw
	TempValue0=Object.PropertyValue
	TempValue0<<=1
	TempValue0+=Object.Value1
	DrawSprite(TempValue0)
endsub
