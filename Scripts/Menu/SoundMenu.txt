#alias 11: TYPE_SOUNDTRACKMENU


sub ObjectMain
        Object.Alpha=Object[-10].Alpha
	switch Object.State
	case 0
#platform: Standard
		Object.Value1=Options.Soundtrack
#endplatform
#platform: Mobile
		if Options.PhysicalControls==0
			Object.Value1=-1
		else
			Object.Value1=Options.Soundtrack
		endif
#endplatform
		if Object.Alpha>254
			Object.State++
		endif
		break
	case 1
#platform: Mobile
		if Options.PhysicalControls==1
#endplatform
			if KeyPress.Left==1
				Object.Value1--
				if Object.Value1<0
					Object.Value1=1
				endif
				PlaySfx(23,0)
			endif
			if KeyPress.Right==1
				Object.Value1++
				if Object.Value1>1
					Object.Value1=0
				endif
				PlaySfx(23,0)
			endif
#platform: Mobile
			CheckTouchRect(0,0,Screen.XSize,Screen.YSize)
			if CheckResult>-1
				Options.PhysicalControls=0
				Object.Value1=-1
			endif
		else
			CheckTouchRect(0,0,Screen.XSize,Screen.YSize)
			TempValue3=CheckResult
			TempValue0=Screen.CenterX
			TempValue0-=90
			TempValue1=Screen.CenterX
			TempValue1-=22
			CheckTouchRect(TempValue0,112,TempValue1,144)
			if CheckResult>-1
				Object.Value1=0
			else
				if TempValue3<0
					if Object.Value1==0
						KeyPress.Start=1
					endif
				else
					if Object.Value1==0
						Object.Value1=-1
					endif
				endif
			endif
			TempValue0=Screen.CenterX
			TempValue0+=22
			TempValue1=Screen.CenterX
			TempValue1+=90
			CheckTouchRect(TempValue0,112,TempValue1,144)
			if CheckResult>-1
				Object.Value1=1
			else
				if TempValue3<0
					if Object.Value1==1
						KeyPress.Start=1
					endif
				else
					if Object.Value1==1
						Object.Value1=-1
					endif
				endif
			endif
			CallFunction(MenuHeading_Function0)
			if KeyPress.Left==1
				Object.Value1=1
				Options.PhysicalControls=1
			endif
			if KeyPress.Right==1
				Object.Value1=0
				Options.PhysicalControls=1
			endif
		endif
#endplatform
		CheckResult=0
#platform: Mobile				
		if Object.Value1>-1
#endplatform
			if KeyPress.Start==1
				CheckResult=1
			endif
			if KeyPress.ButtonA==1
				CheckResult=1
			endif
#platform: Mobile	
		endif
#endplatform
		if CheckResult==1
			Options.Soundtrack=Object.Value1
			SaveRAM[38]=Options.Soundtrack
			Object.State++
			StopMusic()
			PlaySfx(27,0)
			WriteSaveRAM()
		endif
		if KeyPress.ButtonB==1
			Object.Value0=248
			Object.State=3
			PlayStageSfx(1,0)
		endif
		break
	case 2
		if Object.AnimationTimer<30
			Object.AnimationTimer++
		else
			Object.AnimationTimer=0
			Object.State=3
			PlayMusic(Options.Soundtrack)
		endif
		break
	case 3
		if Object.Value0>0
			Object.Alpha=Object.Value0
			Object.Value0-=8
		endif
		break

	endswitch
endsub


sub ObjectDraw
	TempValue0=Object.Value2
	TempValue0>>=5
	TempValue3=Screen.CenterY
	TempValue3+=24
	DrawSpriteScreenFX(TempValue0,FX_INK,Screen.CenterX,TempValue3)
	Object.Value2++
	Object.Value2&=63
	Sin(TempValue0,Object.Value3)
	TempValue0>>=6
	TempValue0+=Screen.CenterY
	TempValue0+=24
	DrawSpriteScreenFX(6,FX_INK,Screen.CenterX,TempValue0)
	Sin(TempValue0,Object.Value3)
	TempValue0>>=6
	FlipSign(TempValue0)
	TempValue0+=Screen.CenterY
	TempValue0+=24
	DrawSpriteScreenFX(7,FX_INK,Screen.CenterX,TempValue0)
	Object.Value3+=2
	Object.Value3&=511
	TempValue0=Screen.CenterX
	TempValue0-=56
	TempValue1=Screen.CenterX
	TempValue1+=56
	TempValue3=Screen.CenterY
	TempValue3+=32
	if Object.Value1==0
		TempValue2=Object.AnimationTimer
		TempValue2&=3
		TempValue2>>=1
		TempValue2+=2
		DrawSpriteScreenFX(TempValue2,FX_INK,TempValue0,TempValue3)
	else
		DrawSpriteScreenFX(3,FX_INK,TempValue0,TempValue3)
	endif
	if Object.Value1==1
		TempValue2=Object.AnimationTimer
		TempValue2&=3
		TempValue2>>=1
		TempValue2+=4
		DrawSpriteScreenFX(TempValue2,FX_INK,TempValue1,TempValue3)
	else
		DrawSpriteScreenFX(5,FX_INK,TempValue1,TempValue3)
	endif
	if Options.Soundtrack==0
		DrawSpriteScreenFX(8,FX_INK,TempValue0,TempValue3)
	else
		DrawSpriteScreenFX(8,FX_INK,TempValue1,TempValue3)
	endif
endsub
