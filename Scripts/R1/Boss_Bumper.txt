sub ObjectMain
	switch Object.State
	case 0
		Object.State=1
		ArrayPos0=Object.Value0
		Object.XPos=Object[ArrayPos0].XPos
		Object.XPos-=1245184
		Object.YPos=Object[ArrayPos0].YPos
		if Object[ArrayPos0].Frame!=0
			if Object[ArrayPos0].Frame==1
				Object.XPos+=524288
			else
				Object.XPos+=1048576
			endif
		endif
		break
	case 1
		ArrayPos0=Object.Value0
		Object.XPos=Object[ArrayPos0].XPos
		Object.XPos-=1245184
		Object.YPos=Object[ArrayPos0].YPos
		if Object[ArrayPos0].Frame!=0
			if Object[ArrayPos0].Frame==1
				Object.XPos+=524288
			else
				Object.XPos+=1048576
			endif
		endif
		if Object.Frame>0
			Object.Frame--
		endif
		break
	case 2
		Object.Value2-=1280
		Object.Value5+=6144
		Object.XPos+=Object.Value2
		Object.YPos+=Object.Value5
		if Object.OutOfBounds==1
			Object.Type=TypeName[BlankObject]
			Object.Priority=0
		endif
		if Object.Frame>0
			Object.Frame--
		endif
		break

	endswitch
endsub

					Player.Speed=-262144
					Player.XVelocity=-262144
					if Object.Frame==0

					endif

#platform: Use_Haptics

		endif
	endif
endsub
