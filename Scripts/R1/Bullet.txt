sub ObjectMain
	Object.Value2+=8192
	Object.XPos+=Object.Value1
	Object.YPos+=Object.Value2
	if Object.OutOfBounds==1
		Object.Type=TypeName[BlankObject]
	else
		TempValue0=Screen.YOffset
		TempValue0-=8
		TempValue0<<=16
		if Object.YPos<TempValue0
			Object.Type=TypeName[BlankObject]
		endif
	endif
	Object.Value0++
	if Object.Value0>2
		Object.Value0=0
		Object.Frame++
		Object.Frame&=3
	endif
endsub
