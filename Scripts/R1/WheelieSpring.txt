sub ObjectMain
	if Object.State==0
		if Object.Value0<Object.PropertyValue
			Object.Value0++
			Object.XPos+=98304
		else
			Object.State=1
			Object.Value0=Object.PropertyValue
			FlipSign(Object.Value0)
		endif
	else
		if Object.Value0<Object.PropertyValue
			Object.Value0++
			Object.XPos-=98304
		else
			Object.State=0
			Object.Value0=Object.PropertyValue
			FlipSign(Object.Value0)
		endif
	endif
	if Object.Frame!=0
		if Object.Frame>12
			Object.Frame=0
		else
			Object.Frame++
		endif
	endif
	ObjectTileGrip(0,0,15,0)
endsub
