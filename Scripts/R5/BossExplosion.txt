#alias 65: TYPE_BOSSEXPLOSION


sub ObjectMain
	switch Object.State
	case 0
		Object.State++
		break
	case 1
		Object.YPos-=196608
		Object.Value0++
		if Object.Value0>3
			Object.Value0=0
			Object.Frame++
			if Object.Frame>4
				Object.Frame=0
				Object.Value1++
				if Object.Value1==2
					Object.Type=TypeName[BlankObject]
				endif
			endif
		endif
		break
	endswitch
endsub


sub ObjectDraw
	DrawSprite(Object.Frame)
endsub
