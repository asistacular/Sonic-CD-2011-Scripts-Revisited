sub ObjectMain
	switch Object.State
	case 0
		Player.XPos=Object[-1].XPos
		Player.Speed=0
		if Object.Value0<120
			Object.Value0++
		else
			Player.Score+=1000
			Object.Value0=0
			Object.Priority=1
			Object.Value1=32768
			Object.State++
		endif
		break
	case 1
		Object.Value1+=3072
		Object.XPos+=Object.Value1
		if Object.OutOfBounds==1
			ResetObjectEntity(Object.EntityNo,TypeName[BlankObject],0,0,0)
		endif
		break
	endswitch
	Object.Frame++
	Object.Frame%=24
	Object.AnimationTimer++
	Object.AnimationTimer%=12
endsub


sub ObjectDraw
	TempValue0=Object.Frame
	TempValue0/=6
	DrawSprite(TempValue0)
	TempValue0=Object.AnimationTimer
	TempValue0/=6
	TempValue0+=4
	TempValue1=Object.AnimationTimer
	TempValue1%=6
	if TempValue1<3
		DrawSprite(TempValue0)
	endif
endsub
