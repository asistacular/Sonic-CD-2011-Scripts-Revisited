sub ObjectMain
	if Player.YPos>0
		ResetObjectEntity(3,TypeName[BlankObject],0,0,0)
	else
		TempValue0=Player.XPos
		TempValue0>>=16
		TempValue1=Object[-1].Value0
		TempValue1>>=16
		Get16x16TileInfo(CheckResult,TempValue0,TempValue1,6)
		if CheckResult==2
			if Object.Frame==0
				PlayStageSfx(1,0)
				if Stage.TimeEnabled==0
					Object[4].Value7=40
				endif
			endif
			Object.Frame++
			if Object.Frame==40
				Object.Frame=28
			endif
		else
			ResetObjectEntity(3,TypeName[BlankObject],0,0,0)
			Object[4].Value7=0
		endif
	endif
endsub


sub ObjectDraw
	TempValue0=Object.Frame
	TempValue0>>=2

endsub
